class ChessVision(object):
    def __init__(self, model_file=None, device=0, board_size=(400,400)):
        self.capture = cv2.VideoCapture(device)
        if not self.capture.isOpened():
            raise RuntimeError('Failed to start camera.')

        self.capture.set(3,1280)
        self.capture.set(4,1024)
        self.board_area = 185

        self.contour = None
        self.dwell_states = deque(maxlen=20)
        
        self.board_size = board_size
        self.square_size = tuple(np.asarray(self.board_size)//8)

        self.label_map = list('KQRBNP_kqrbnp')
        
        if model_file:
            if isinstance(model_file, str):
                self.model = load_model(model_file)
            else:
                self.model = model_file
        else:
            self.model = None

    def dewarp(self, img):
        pts = self.contour.reshape(4, 2)
        rect = np.zeros((4, 2), dtype='float32')

        s = pts.sum(axis=1)
        rect[0] = pts[np.argmin(s)]
        rect[2] = pts[np.argmax(s)]

        diff = np.diff(pts, axis=1)
        rect[1] = pts[np.argmin(diff)]
        rect[3] = pts[np.argmax(diff)]

        (tl, tr, br, bl) = rect
        f = lambda a,b: np.sqrt(
                ((a[0]-b[0])**2)+((a[1]-b[1])**2))
        mw = max(int(f(br, bl)), int(f(tr, tl)))
        mh = max(int(f(tr, br)), int(f(tl, bl)))

        dst = np.array([[0, 0],[mw - 1, 0],[mw - 1, mh - 1],[0, mh - 1]],
                        dtype='float32')

        M = cv2.getPerspectiveTransform(rect, dst)
        img = cv2.warpPerspective(img, M, (mw, mh))

        return cv2.resize(img, self.board_size)


    def find_board(self, img):
        if self.contour is not None:
            return self.dewarp(img)

        gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        edged = cv2.Canny(gray, 30, 200)
        _, contours, _ = cv2.findContours(edged, 1, 2)
        contours = sorted(contours, key=cv2.contourArea, reverse=True)[:3]

        for contour in contours[::-1]:
            approx = cv2.approxPolyDP(
                contour, 0.05*cv2.arcLength(contour,True),True)
            area = cv2.contourArea(approx) // 1e3
            #print(area)
            if len(approx) == 4 and area > 180 and area < 200: #area == self.board_area:
                self.contour = approx
                break
                #cv2.drawContours(img, [approx], -1, (0, 255, 0), 3)

        if self.contour is not None:
            return self.dewarp(img)
        return None

    def eval_square(self, img, row, col, debug=False):
        w,h = img.shape[:2]
        s = w//8
        x1,y1,x2,y2 = row*s,col*s,row*s+s,col*s+s
        square = img[y1:y2,x1:x2]
        prediction = self.predict(square)
        if debug and prediction != '_':
            color = (0,0,255)
            if prediction in string.ascii_uppercase:
                color = (255,0,0)
            cv2.putText(square ,prediction, (s//3,s-10), 0, 1, color, 2)
            cv2.rectangle(img,(x1,y1),(x2,y2),color ,2)
        return square,prediction

    def predict(self, img, plot=False):
        if self.model is None:
            return ''
        
        img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        img = cv2.resize(img, self.square_size)
        img = img.reshape(1, *self.square_size, 1) / 255
        pred = self.model.predict(img)
        return self.label_map[np.argmax(pred)]


    def get_board_matrix(self, img, debug=False):
        mat = []
        for i in range(8):
            for j in range(8):
                s,p = self.eval_square(img, j, i, debug)
                mat.append(p)
        return np.asarray(mat)
    
    def to_fen(self, mat):
        def f(s):
            for p,c in [
                [k, len(list(g))] for k,g in groupby(s)]:
                if p == '_':
                    yield str(c)
                else:
                    yield p * c
        for row in mat.reshape(8,8):
            s = ''.join(list(f(''.join(row))))
            yield s

    def run(self, debug=False):
        while True:
            ret, img = self.capture.read()
            if img is None:
                raise RuntimeError('Failed to read frame.')

            board = self.find_board(img)
            main = cv2.resize(img, (800,450))

            if board is None:
                cv2.imshow('',main) 
            else:
                mat = self.get_board_matrix(board, debug)
                fen = '/'.join(list(self.to_fen(mat))) + ' w KQkq - 0 1'
                fh = io.BytesIO()
                try:
                    svg_data = chess.svg.board(
                        board=chess.Board(fen),
                        size=self.board_size[0], 
                        coordinates=False)
                except:
                    svg_data = chess.svg.board(
                        board=chess.Board(),
                        size=self.board_size[0], 
                        coordinates=False)

                cairosvg.svg2png(svg_data, write_to=fh)
                data = np.asarray(bytearray(fh.getvalue()), dtype=np.uint8)
                board2d = cv2.imdecode(data, cv2.IMREAD_COLOR)
                boards = np.hstack([board2d, board])
                
                final = np.vstack([boards,main])
                cv2.imshow('',final) 

            if cv2.waitKey(25) & 0xFF == ord('q'):
                cv2.destroyAllWindows()
                self.capture.release()
                break
